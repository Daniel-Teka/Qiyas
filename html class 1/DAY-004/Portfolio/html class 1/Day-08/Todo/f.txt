function getValue() {
  const num1 = document.getElementById("num1").value;

  let history = [];

  // console.log("out side the condition")
  if (!num1 || !num2 || !operator) {
    //console.log("inside the condition")
    result1 = "Num1, Num2 and Operator should not empty";
    result.innerText = result1;
    return;
  }
  if (operator == "/") {
    result1 = Number(num1) / Number(num2);
  } else if (operator == "+") {
    result1 = Number(num1) + Number(num2);
  } else if (operator == "-") {
    result1 = Number(num1) - Number(num2);
  } else if (operator == "*") {
    result1 = Number(num1) * Number(num2);
  } else {
    result1 = "Invalid Operator";
  }

  result.innerText = result1;

  push({
    num1: num1,
    num2: num2,
    operator: operator,
    result: result1,
  });
  let ul = document.createElement("ul");
  for (let i = 0; /* 0<1 */    i < history.length; i++) {
    const element = history[i];
    const node = document.createElement("li");
    const textnode = document.createTextNode(
     /* .key */ element.num1 + element.operator + element.num2 + " =" + element.result,
    );
    node.appendChild(/*the result*/textnode);
    document.getElementById("ul").appendChild(node);
  }
}
